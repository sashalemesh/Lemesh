___________________________________________Определяем настройки кастомайзера. Описываем секции, настройки и контролы____________________________
function mytheme_customize_register( $wp_customize ) {
//В этой настройке сохраняем заголовок в футере
   $wp_customize->add_setting( 'header_textcolor' , array(
	/*
         * Вставляем текст по умолчанию и добавляем метод
         * с помощью которого мы сможем переводить текс
         */
    'default'     => '#000000',
	/*
         * метод с помощью которо смотрим изменения в кастомайзере
         * Еще есть метод 'postMessage' для java script
         */
    'transport'   => 'refresh',
) );
	//Вставляем еще секцию. Указываем id секции
	$wp_customize->add_section( 'mytheme_new_section_name' , array(
	//Два нижних подчеркивания функция для перевода
    'title'      => __( 'Visible Section Name', 'mytheme' ),
    'priority'   => 30,
) );
 	//Добавляем контроллер
	$wp_customize->add_control(
	'your_control_id', 
	array(
		'label'    => __( 'Control Label', 'mytheme' ),
		'section'  => 'your_section_id',
		'settings' => 'your_setting_id',
		'type'     => 'radio',
		'choices'  => array(
			'left'  => 'left',
			'right' => 'right',
		),
	)
);
}
add_action( 'customize_register', 'mytheme_customize_register' );

    ) );
